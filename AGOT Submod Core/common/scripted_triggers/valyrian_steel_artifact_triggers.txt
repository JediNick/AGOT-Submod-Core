portrait_has_unique_sword_trigger = {
		any_equipped_character_artifact = {
			artifact_type = sword   						
			OR = {
				has_variable = vs_sword_1
				has_variable = vs_sword_2
				has_variable = vs_sword_3
				
			}	
		}
	}
